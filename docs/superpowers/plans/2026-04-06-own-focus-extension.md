# OWN-Focus Chrome Extension Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build OWN-Focus, an all-in-one Chrome MV3 side panel extension for students with 8 tools: AI Chat, Desmos, YouTube, PDF Viewer, Pomodoro, Stopwatch/Timer, Notes, and Lock In Mode.

**Architecture:** Single side panel HTML file (`sidepanel.html`) houses all tool views as hidden `<div>` sections; `sidepanel.js` handles navigation by toggling `display` without re-rendering. Each tool's logic lives in its own `tools/*.js` file loaded conditionally. Background service worker (`background.js`) handles chrome.alarms for timers and tab-monitoring for Lock In Mode.

**Tech Stack:** Vanilla HTML/CSS/JS (MV3), Chrome Extension APIs (sidePanel, storage, tabs, scripting, alarms, notifications, windows), PDF.js (bundled), YouTube IFrame API + Invidious for search, Web Audio API for chime sounds.

```javascript
// background.js — Service worker

// Open side panel when extension icon is clicked
chrome.action.onClicked.addListener((tab) => {
  chrome.sidePanel.open({ windowId: tab.windowId });
});

// ── Alarm handling (Pomodoro + Countdown Timer) ──────────────────────────────
chrome.alarms.onAlarm.addListener((alarm) => {
  if (alarm.name.startsWith('pomodoro-')) {
    handlePomodoroAlarm(alarm.name);
  } else if (alarm.name === 'countdown-end') {
    handleCountdownAlarm();
  }
});

function handlePomodoroAlarm(name) {
  const phase = name.replace('pomodoro-', '');
  chrome.notifications.create('pomodoro-notify', {
    type: 'basic',
    iconUrl: 'icons/icon48.png',
    title: 'OWN-Focus',
    message: phase === 'work' ? 'Focus time! Get to work.' : 'Break time! Rest up.',
    silent: false
  });
  // Signal the side panel if open
  chrome.storage.local.set({ pomodoroAlarmFired: { phase, ts: Date.now() } });
}

function handleCountdownAlarm() {
  chrome.notifications.create('countdown-notify', {
    type: 'basic',
    iconUrl: 'icons/icon48.png',
    title: 'OWN-Focus Timer',
    message: "Time's up!",
    silent: false
  });
  chrome.storage.local.set({ countdownAlarmFired: Date.now() });
}

// ── Lock In Mode: tab switch detection ───────────────────────────────────────
let lockInState = { active: false, mode: null, lockedTabId: null, lockedWindowId: null };

chrome.storage.local.get('lockInState', (result) => {
  if (result.lockInState) lockInState = result.lockInState;
});

chrome.storage.onChanged.addListener((changes) => {
  if (changes.lockInState) {
    lockInState = changes.lockInState.newValue || { active: false };
  }
});

chrome.tabs.onActivated.addListener(async (activeInfo) => {
  if (!lockInState.active) return;
  if (activeInfo.tabId === lockInState.lockedTabId) return;

  if (lockInState.mode === 'soft') {
    // Inject overlay into the newly activated tab
    try {
      await chrome.scripting.executeScript({
        target: { tabId: activeInfo.tabId },
        files: ['content.js']
      });
      await chrome.scripting.insertCSS({
        target: { tabId: activeInfo.tabId },
        files: ['content.css']
      });
      await chrome.scripting.executeScript({
        target: { tabId: activeInfo.tabId },
        func: () => { if (window.__ownFocusShowSoftOverlay) window.__ownFocusShowSoftOverlay(); }
      });
    } catch (e) {
      // Tab may not allow scripting (chrome:// pages etc.)
    }
  } else if (lockInState.mode === 'hard') {
    // Switch back to locked tab
    try {
      await chrome.tabs.update(lockInState.lockedTabId, { active: true });
    } catch (e) {
      // Tab was closed — end lock in
      lockInState = { active: false };
      chrome.storage.local.set({ lockInState });
    }
  }
});

chrome.windows.onFocusChanged.addListener((windowId) => {
  if (!lockInState.active || lockInState.mode !== 'hard') return;
  if (windowId === chrome.windows.WINDOW_ID_NONE) return;
  if (windowId !== lockInState.lockedWindowId) {
    chrome.windows.update(lockInState.lockedWindowId, { focused: true });
  }
});
```

// ── Soft Lock Overlay ─────────────────────────────────────────────────────────
window.__ownFocusShowSoftOverlay = function() {
  if (document.getElementById('own-focus-overlay')) return;
  const overlay = document.createElement('div');
  overlay.id = 'own-focus-overlay';
  overlay.innerHTML = `
    <div class="own-focus-overlay-box">
      <div class="own-focus-overlay-icon">🔒</div>
      <h2>You're in Focus Mode</h2>
      <p>Get back to work!</p>
      <div class="own-focus-overlay-btns">
        <button id="own-focus-return">Return to Study</button>
        <button id="own-focus-break">Take a Break</button>
      </div>
    </div>
  `;
  document.body.appendChild(overlay);

  document.getElementById('own-focus-break').onclick = () => overlay.remove();
  document.getElementById('own-focus-return').onclick = () => {
    overlay.remove();
    chrome.runtime.sendMessage({ type: 'LOCK_IN_RETURN' });
  };
};

// ── PDF Split Screen ──────────────────────────────────────────────────────────
window.__ownFocusShowSplitScreen = function(pdfDataUrl) {
  if (document.getElementById('own-focus-split')) return;

  const original = document.documentElement.innerHTML;
  document.documentElement.style.overflow = 'hidden';

  const container = document.createElement('div');
  container.id = 'own-focus-split';
  container.innerHTML = `
    <div id="own-focus-split-left">
      <iframe id="own-focus-original-frame" src="${location.href}"></iframe>
    </div>
    <div id="own-focus-split-divider"></div>
    <div id="own-focus-split-right">
      <iframe id="own-focus-pdf-frame" src="${pdfDataUrl}"></iframe>
    </div>
    <button id="own-focus-split-close">✕ Close Split</button>
  `;

  document.body.style.display = 'none';
  document.documentElement.appendChild(container);

  // Drag to resize
  const divider = document.getElementById('own-focus-split-divider');
  let dragging = false;
  divider.addEventListener('mousedown', () => { dragging = true; });
  document.addEventListener('mousemove', (e) => {
    if (!dragging) return;
    const pct = (e.clientX / window.innerWidth) * 100;
    document.getElementById('own-focus-split-left').style.width = pct + '%';
    document.getElementById('own-focus-split-right').style.width = (100 - pct - 0.5) + '%';
  });
  document.addEventListener('mouseup', () => { dragging = false; });

  document.getElementById('own-focus-split-close').onclick = () => {
    container.remove();
    document.body.style.display = '';
  };
};

```html
<!DOCTYPE html>
<html>
<head><title>Generate OWN-Focus Icons</title></head>
<body>
<button onclick="generateAll()">Download All Icons</button>
<script>
function generateIcon(size) {
  const canvas = document.createElement('canvas');
  canvas.width = size; canvas.height = size;
  const ctx = canvas.getContext('2d');
  // Background
  ctx.fillStyle = '#1a1a2e';
  ctx.beginPath();
  ctx.roundRect(0, 0, size, size, size * 0.18);
  ctx.fill();
  // Amber "O" letter
  ctx.fillStyle = '#F59E0B';
  ctx.font = `bold ${size * 0.58}px system-ui`;
  ctx.textAlign = 'center';
  ctx.textBaseline = 'middle';
  ctx.fillText('O', size / 2, size / 2);
  return canvas;
}

function generateAll() {
  [16, 48, 128].forEach(size => {
    const canvas = generateIcon(size);
    const link = document.createElement('a');
    link.download = `icon${size}.png`;
    link.href = canvas.toDataURL('image/png');
    link.click();
  });
}
</script>
</body>
</html>
```

  next.classList.add('active');
  currentView = toolId;

// tools/notes.js — Multi-note editor with auto-save

const NOTES_STORAGE_KEY = 'ownFocusNotes';
const NOTES_ACTIVE_KEY = 'ownFocusActiveNote';

let notes = {}; // { id: { id, title, content, createdAt, updatedAt } }
let activeNoteId = null;

// ── Storage helpers ──────────────────────────────────────────────────────────
function saveNotes() {
  chrome.storage.local.set({ [NOTES_STORAGE_KEY]: notes }, () => {
    if (chrome.runtime.lastError) {
      const bytes = JSON.stringify(notes).length;
      if (bytes > 4 * 1024 * 1024) { // ~4MB warning
        showStorageWarning();
      }
    }
  });
}

function loadNotes(cb) {
  chrome.storage.local.get([NOTES_STORAGE_KEY, NOTES_ACTIVE_KEY], (result) => {
    notes = result[NOTES_STORAGE_KEY] || {};
    activeNoteId = result[NOTES_ACTIVE_KEY] || null;
    cb();
  });
}

function showStorageWarning() {
  const warn = document.createElement('div');
  warn.style.cssText = 'position:fixed;top:10px;left:10px;right:10px;background:#EF4444;color:white;padding:10px;border-radius:8px;font-size:13px;z-index:9999;text-align:center';
  warn.textContent = 'Storage nearly full — export old notes to free space.';
  document.body.appendChild(warn);
  setTimeout(() => warn.remove(), 5000);
}

// ── Note CRUD ────────────────────────────────────────────────────────────────
function createNote() {
  const id = 'note_' + Date.now();
  const now = Date.now();
  notes[id] = { id, title: 'Untitled Note', content: '', createdAt: now, updatedAt: now };
  saveNotes();
  chrome.storage.local.set({ [NOTES_ACTIVE_KEY]: id });
  return id;
}

function deleteNote(id) {
  if (!confirm(`Delete "${notes[id]?.title}"?`)) return;
  delete notes[id];
  saveNotes();
  const remaining = Object.keys(notes);
  if (remaining.length === 0) {
    activeNoteId = createNote();
  } else {
    activeNoteId = remaining[remaining.length - 1];
    chrome.storage.local.set({ [NOTES_ACTIVE_KEY]: activeNoteId });
  }
  renderNotesList();
  loadActiveNote();
}

function renameNote(id) {
  const newTitle = prompt('Note name:', notes[id]?.title);
  if (!newTitle || !newTitle.trim()) return;
  notes[id].title = newTitle.trim();
  notes[id].updatedAt = Date.now();
  saveNotes();
  renderNotesList();
}

function duplicateNote(id) {
  const source = notes[id];
  const newId = 'note_' + Date.now();
  notes[newId] = { ...source, id: newId, title: source.title + ' (copy)', createdAt: Date.now(), updatedAt: Date.now() };
  saveNotes();
  renderNotesList();
}

// ── Render notes list ────────────────────────────────────────────────────────
function renderNotesList(filter = '') {
  const list = document.getElementById('notes-list');
  list.innerHTML = '';

  const filtered = Object.values(notes)
    .filter(n => !filter || n.title.toLowerCase().includes(filter.toLowerCase()) || n.content.toLowerCase().includes(filter.toLowerCase()))
    .sort((a, b) => b.updatedAt - a.updatedAt);

  filtered.forEach(note => {
    const item = document.createElement('div');
    item.className = 'note-list-item' + (note.id === activeNoteId ? ' active' : '');
    item.textContent = note.title;
    item.title = note.title;

    // Context menu on right-click
    item.addEventListener('contextmenu', (e) => {
      e.preventDefault();
      showNoteContextMenu(note.id, e.clientX, e.clientY);
    });

    item.addEventListener('click', () => {
      saveCurrentNote();
      activeNoteId = note.id;
      chrome.storage.local.set({ [NOTES_ACTIVE_KEY]: activeNoteId });
      renderNotesList(document.getElementById('notes-search').value);
      loadActiveNote();
    });

    list.appendChild(item);
  });
}

// ── Load and save active note ────────────────────────────────────────────────
function loadActiveNote() {

  if (!note) return;
  const editor = document.getElementById('notes-editor');
  editor.innerHTML = note.content;
  updateMeta(note);
}

function saveCurrentNote() {
  if (!activeNoteId || !notes[activeNoteId]) return;
  const editor = document.getElementById('notes-editor');
  notes[activeNoteId].content = editor.innerHTML;
  notes[activeNoteId].updatedAt = Date.now();

  // Update title from first line if untitled
  if (notes[activeNoteId].title === 'Untitled Note') {
    const text = editor.innerText.trim().split('\n')[0].slice(0, 40);
    if (text) notes[activeNoteId].title = text;
  }
  saveNotes();
}

function updateMeta(note) {
  const meta = document.getElementById('notes-meta');
  const fmt = (ts) => new Date(ts).toLocaleDateString('en-US', { month: 'short', day: 'numeric', hour: '2-digit', minute: '2-digit' });

// ── Toolbar commands ─────────────────────────────────────────────────────────
function setupToolbar() {
  document.getElementById('notes-toolbar').addEventListener('click', (e) => {
    const btn = e.target.closest('[data-cmd]');
    if (btn) {
      e.preventDefault();
      document.execCommand(btn.dataset.cmd, false, null);
      document.getElementById('notes-editor').focus();
      return;
    }

    const hBtn = e.target.closest('[data-heading]');
    if (hBtn) {
      e.preventDefault();
      document.execCommand('formatBlock', false, hBtn.dataset.heading);

      if (color === 'transparent') {
        document.execCommand('removeFormat', false, null);
      } else {
        document.execCommand('hiliteColor', false, color);
      }
      document.getElementById('notes-editor').focus();
    }
  });
}

// ── Export ────────────────────────────────────────────────────────────────────
function exportNote(format) {
  const note = notes[activeNoteId];
  if (!note) return;

  let content, filename, mime;
  if (format === 'txt') {
    const tmp = document.createElement('div');
    tmp.innerHTML = note.content;
    content = tmp.innerText;
    filename = note.title + '.txt';
    mime = 'text/plain';
  } else {
    // Markdown: basic conversion
    const tmp = document.createElement('div');
    tmp.innerHTML = note.content;
    content = htmlToMarkdown(tmp);
    filename = note.title + '.md';
    mime = 'text/markdown';
  }

  const blob = new Blob([content], { type: mime });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = filename;
  a.click();
  URL.revokeObjectURL(url);
}

function htmlToMarkdown(el) {
  let md = '';
  el.childNodes.forEach(node => {
    if (node.nodeType === Node.TEXT_NODE) {
      md += node.textContent;
    } else if (node.nodeName === 'H1') {
      md += '# ' + node.innerText + '\n\n';
    } else if (node.nodeName === 'H2') {
      md += '## ' + node.innerText + '\n\n';
    } else if (node.nodeName === 'H3') {
      md += '### ' + node.innerText + '\n\n';
    } else if (node.nodeName === 'STRONG' || node.nodeName === 'B') {
      md += '**' + node.innerText + '**';
    } else if (node.nodeName === 'EM' || node.nodeName === 'I') {
      md += '_' + node.innerText + '_';
    } else if (node.nodeName === 'CODE') {
      md += '`' + node.innerText + '`';
    } else if (node.nodeName === 'UL') {
      node.querySelectorAll('li').forEach(li => { md += '- ' + li.innerText + '\n'; });
      md += '\n';
    } else if (node.nodeName === 'OL') {
      node.querySelectorAll('li').forEach((li, i) => { md += `${i+1}. ${li.innerText}\n`; });
      md += '\n';
    } else if (node.nodeName === 'P' || node.nodeName === 'DIV' || node.nodeName === 'BR') {
      md += (node.innerText || '') + '\n';
    } else {
      md += node.innerText || '';
    }
  });
  return md;
}

// ── Init ──────────────────────────────────────────────────────────────────────
function initNotes() {
  loadNotes(() => {
    if (Object.keys(notes).length === 0) {
      activeNoteId = createNote();
    }
    if (!activeNoteId || !notes[activeNoteId]) {

        saveCurrentNote();
        renderNotesList(document.getElementById('notes-search').value);
      }, 300);
    });

    // New note button
    document.getElementById('notes-new-btn').addEventListener('click', () => {
      saveCurrentNote();
      activeNoteId = createNote();
      chrome.storage.local.set({ [NOTES_ACTIVE_KEY]: activeNoteId });
      renderNotesList();
      loadActiveNote();
      editor.focus();
    });

    // Search
    document.getElementById('notes-search').addEventListener('input', (e) => {
      renderNotesList(e.target.value);
    });

    // Export buttons
    document.getElementById('notes-export-md').addEventListener('click', () => exportNote('md'));
    document.getElementById('notes-export-txt').addEventListener('click', () => exportNote('txt'));
  });
}

const POMO_KEY = 'ownFocusPomodoro';

// Phase sequences: 4 work phases separated by 3 short breaks + 1 long break
const DEFAULT_SETTINGS = { workMins: 25, shortMins: 5, longMins: 15 };

let pomState = {
  running: false,
  phase: 'work',          // 'work' | 'short' | 'long'
  cycle: 0,               // 0-3 (which of the 4 work blocks we're on)
  endTime: null,          // epoch ms when current phase ends
  totalPausedMs: 0,       // accumulated paused time
  pausedAt: null,         // epoch ms when last paused
  settings: { ...DEFAULT_SETTINGS },
  todayFocusMs: 0,
  todayDate: new Date().toDateString(),

let pomTick = null;

// ── Persist/restore ───────────────────────────────────────────────────────────
function savePom() {

function loadPom(cb) {
  chrome.storage.local.get(POMO_KEY, (result) => {
    if (result[POMO_KEY]) {
      pomState = result[POMO_KEY];

}

// ── Phase helpers ─────────────────────────────────────────────────────────────
function getPhaseDurationMs() {
  const s = pomState.settings;
  if (pomState.phase === 'work') return s.workMins * 60000;
  if (pomState.phase === 'short') return s.shortMins * 60000;
  return s.longMins * 60000;
}

function getPhaseLabel() {
  if (pomState.phase === 'work') return 'Focus Time';
  if (pomState.phase === 'short') return 'Short Break';
  return 'Long Break';
}

function getPhaseColor() {
  return pomState.phase === 'work' ? '#F59E0B' : '#10B981';
}

function nextPhase() {
  if (pomState.phase === 'work') {
    // Accumulate focus time
    if (pomState.lastPhaseStartMs) {
      pomState.todayFocusMs += Date.now() - pomState.lastPhaseStartMs;
    }
    if (pomState.cycle < 3) {
      pomState.phase = 'short';
    } else {
      pomState.phase = 'long';
      pomState.cycle = 0;
    }
  } else {
    pomState.phase = 'work';
    if (pomState.phase === 'work' && pomState.cycle < 3) pomState.cycle++;
    // correct: after a break we move to next work cycle
    if (pomState.cycle < 3) pomState.cycle++;
    else pomState.cycle = 0;
    pomState.lastPhaseStartMs = Date.now();
  }
  pomState.running = false;
  pomState.endTime = null;
  pomState.pausedAt = null;
  pomState.totalPausedMs = 0;
  savePom();
}

// Fix: simpler cycle progression
function advancePhase() {
  // Accumulate focus if we were in work phase
  if (pomState.phase === 'work' && pomState.lastPhaseStartMs) {
    pomState.todayFocusMs += Date.now() - pomState.lastPhaseStartMs;
    pomState.lastPhaseStartMs = null;
  }

  if (pomState.phase === 'work') {
    pomState.cycle++;
    if (pomState.cycle >= 4) {
      pomState.phase = 'long';
      pomState.cycle = 4; // mark full set done
    } else {
      pomState.phase = 'short';
    }
  } else if (pomState.phase === 'short') {
    pomState.phase = 'work';
  } else if (pomState.phase === 'long') {
    pomState.phase = 'work';
    pomState.cycle = 0;
  }

  pomState.running = false;
  pomState.endTime = null;
  pomState.pausedAt = null;
  pomState.totalPausedMs = 0;
  savePom();
  updatePomUI();
}

// ── UI update ─────────────────────────────────────────────────────────────────
function updatePomUI() {
  const totalMs = getPhaseDurationMs();
  let remainingMs;

  if (!pomState.running && !pomState.endTime) {
    remainingMs = totalMs;
  } else if (pomState.pausedAt) {
    remainingMs = pomState.endTime - pomState.pausedAt;
  } else {
    remainingMs = Math.max(0, pomState.endTime - Date.now());
  }

  const mins = Math.floor(remainingMs / 60000);
  const secs = Math.floor((remainingMs % 60000) / 1000);
  document.getElementById('pomo-display').textContent =
    `${String(mins).padStart(2, '0')}:${String(secs).padStart(2, '0')}`;

  // Ring progress
  const progress = remainingMs / totalMs;
  const offset = CIRCUMFERENCE * (1 - progress);
  const arc = document.getElementById('pomo-ring-arc');
  arc.style.strokeDashoffset = offset;
  arc.style.stroke = getPhaseColor();

  // Phase label
  document.getElementById('pomo-phase').textContent = getPhaseLabel();
  document.getElementById('pomo-phase').style.color = getPhaseColor();

  // Cycle dots
  const dots = document.querySelectorAll('#pomo-cycle-dots .dot');
  dots.forEach((dot, i) => {
    dot.classList.remove('active', 'done');
    if (pomState.phase === 'work') {
      if (i < pomState.cycle) dot.classList.add('done');
      else if (i === pomState.cycle) dot.classList.add('active');
    } else {
      if (i < pomState.cycle) dot.classList.add('done');
    }
  });

  // Start button label
  document.getElementById('pomo-start').textContent =
    pomState.running ? 'Pause' : (pomState.endTime && pomState.pausedAt ? 'Resume' : 'Start');

  // Today stats
  const focusMins = Math.floor(pomState.todayFocusMs / 60000);
  document.getElementById('pomo-today-time').textContent = focusMins + 'm';

  if (remainingMs <= 0 && pomState.running) {
    // Phase complete
    playChime();
    advancePhase();
  }
}

// ── Controls ──────────────────────────────────────────────────────────────────
function pomStart() {
  if (pomState.running) {
    // Pause
    pomState.pausedAt = Date.now();
    pomState.running = false;
    chrome.alarms.clear('pomodoro-phase');
  } else {
    // Start or Resume
    if (pomState.pausedAt) {
      // Resume: shift endTime forward by pause duration
      const pauseDuration = Date.now() - pomState.pausedAt;
      pomState.endTime += pauseDuration;
      pomState.totalPausedMs += pauseDuration;
      pomState.pausedAt = null;
    } else {
      // Fresh start
      pomState.endTime = Date.now() + getPhaseDurationMs();
      if (pomState.phase === 'work') pomState.lastPhaseStartMs = Date.now();
    }
    pomState.running = true;

    const remainingMs = Math.max(0, pomState.endTime - Date.now());
    chrome.alarms.create('pomodoro-phase', { delayInMinutes: remainingMs / 60000 });
  }
  savePom();
  updatePomUI();
}

function pomReset() {
  chrome.alarms.clear('pomodoro-phase');
  pomState.running = false;
  pomState.endTime = null;
  pomState.pausedAt = null;
  pomState.totalPausedMs = 0;
  pomState.lastPhaseStartMs = null;
  savePom();
  updatePomUI();
}

function pomSkip() {
  chrome.alarms.clear('pomodoro-phase');
  advancePhase();
}

// ── Alarm fired signal ────────────────────────────────────────────────────────
function checkAlarmFired() {
  chrome.storage.local.get('pomodoroAlarmFired', (result) => {
    if (result.pomodoroAlarmFired && result.pomodoroAlarmFired.ts > (Date.now() - 10000)) {
      chrome.storage.local.remove('pomodoroAlarmFired');
      playChime();
      advancePhase();
    }
  });
}

// ── Chime via Web Audio ───────────────────────────────────────────────────────
function playChime() {
  try {
    const ctx = new AudioContext();
    const osc = ctx.createOscillator();
    const gain = ctx.createGain();
    osc.connect(gain);
    gain.connect(ctx.destination);
    osc.frequency.setValueAtTime(880, ctx.currentTime);
    osc.frequency.exponentialRampToValueAtTime(440, ctx.currentTime + 0.4);
    gain.gain.setValueAtTime(0.3, ctx.currentTime);
    gain.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + 0.6);
    osc.start(ctx.currentTime);
    osc.stop(ctx.currentTime + 0.6);
  } catch (e) { /* AudioContext may be blocked */ }
}

// ── Settings ──────────────────────────────────────────────────────────────────
function showPomSettings() {

  document.getElementById('pomodoro-settings').classList.remove('hidden');

  hidePomSettings();

  document.getElementById('pomodoro-settings').classList.add('hidden');
}

// ── Init ──────────────────────────────────────────────────────────────────────
function initPomodoro() {
  loadPom(() => {
    updatePomUI();

    // If timer was running when panel was closed, resume visual
    if (pomState.running && !pomState.pausedAt) {
      pomTick = setInterval(updatePomUI, 1000);

    document.getElementById('pomo-start').addEventListener('click', () => {

    document.getElementById('pomo-save-settings').addEventListener('click', savePomSettings);
    document.getElementById('pomo-cancel-settings').addEventListener('click', hidePomSettings);

    checkAlarmFired();

    // Listen for alarm signals while panel is open
    chrome.storage.onChanged.addListener((changes) => {
      if (changes.pomodoroAlarmFired) checkAlarmFired();
    });
  });

// tools/ai-chat.js — AI Chat with ChatGPT/Claude/Gemini iframes + popup fallback

const AI_LAST_KEY = 'ownFocusLastAI';

const AI_URLS = {
  chatgpt: 'https://chat.openai.com',
  claude:  'https://claude.ai',
  gemini:  'https://gemini.google.com',
};

// Sites known to block iframe embedding — use popup by default
const AI_NO_EMBED = { chatgpt: true, claude: true, gemini: true };

let aiFrames = {};
let activeAI = 'chatgpt';

function loadAiFrame(id) {
  const container = document.getElementById('ai-iframe-container');

  // Hide all existing iframes
  Object.values(aiFrames).forEach(f => { f.style.display = 'none'; });
  document.getElementById('ai-error').classList.add('hidden');

  if (AI_NO_EMBED[id]) {
    // Show error with popup option immediately (these sites always block iframes)
    showAiPopupFallback(id);
    return;
  }

  if (!aiFrames[id]) {
    const iframe = document.createElement('iframe');
    iframe.src = AI_URLS[id];
    iframe.title = id;
    iframe.style.cssText = 'width:100%;height:100%;border:none;display:none';
    iframe.sandbox = 'allow-same-origin allow-scripts allow-forms allow-popups';
    container.appendChild(iframe);
    aiFrames[id] = iframe;

    // Detect X-Frame-Options block (won't fire on CSP, but catches some)
    iframe.addEventListener('load', () => {
      try {
        // If we can access the document, it loaded fine
        const doc = iframe.contentDocument;
        if (!doc || doc.title === '') throw new Error('blocked');
        iframe.style.display = 'block';
      } catch (e) {
        iframe.style.display = 'none';
        showAiPopupFallback(id);
      }
    });
  } else {
    aiFrames[id].style.display = 'block';
  }
}

  const labels = { chatgpt: 'ChatGPT', claude: 'Claude', gemini: 'Gemini' };

  errEl.innerHTML = `
    <p>${labels[id]} blocks embedding in iframes due to security policy.</p>
    <button class="btn-amber" id="ai-popup-btn">Open ${labels[id]} in Popup ↗</button>
  `;
  errEl.classList.remove('hidden');

// ── PDF.js setup ──────────────────────────────────────────────────────────────
function setupPdfJs() {
  if (typeof pdfjsLib === 'undefined') return false;
  pdfjsLib.GlobalWorkerOptions.workerSrc = chrome.runtime.getURL('lib/pdf.worker.min.js');
  return true;

// ── File loading ──────────────────────────────────────────────────────────────
async function loadPdfFile(file) {
  if (!setupPdfJs()) {
    alert('PDF.js library not loaded. Check lib/ directory.');
    return;
  }

  const arrayBuffer = await file.arrayBuffer();
  const loadingTask = pdfjsLib.getDocument({ data: arrayBuffer });

  try {
    pdfDoc = await loadingTask.promise;
    totalPages = pdfDoc.numPages;
    currentPage = 1;

    // Save state
    chrome.storage.local.set({ [PDF_STATE_KEY]: { fileName: file.name, pageCount: totalPages } });

    showPdfControls();
    renderPage(currentPage);

    alert('Failed to load PDF: ' + e.message);
  }

async function renderPage(pageNum) {
  if (!pdfDoc) return;
  const page = await pdfDoc.getPage(pageNum);

  const canvas = document.getElementById('pdf-canvas');
  const ctx = canvas.getContext('2d');
  const viewport = page.getViewport({ scale });

  canvas.height = viewport.height;
  canvas.width = viewport.width;
  canvas.style.maxWidth = '100%';

  await page.render({ canvasContext: ctx, viewport }).promise;

  document.getElementById('pdf-page-info').textContent = `Page ${pageNum} / ${totalPages}`;
  document.getElementById('pdf-zoom-level').textContent = Math.round(scale * 100) + '%';

  chrome.storage.local.set({ [`${PDF_STATE_KEY}_page`]: pageNum });

async function launchSplitScreen(file) {
  if (!setupPdfJs()) return;

  const arrayBuffer = await file.arrayBuffer();
  const base64 = await arrayBufferToBase64(arrayBuffer);
  const dataUrl = 'data:application/pdf;base64,' + base64;

      target: { tabId: tab.id },
      func: (pdfUrl) => {
        if (window.__ownFocusShowSplitScreen) window.__ownFocusShowSplitScreen(pdfUrl);

      args: [dataUrl]

    document.getElementById('pdf-split-drop-zone').classList.add('hidden');

    const blob = new Blob([buffer]);

    reader.readAsDataURL(blob);

}

    if (result[LOCK_HISTORY_KEY]) lockHistory = result[LOCK_HISTORY_KEY];

function formatElapsed(secs) {
  const m = Math.floor(secs / 60);
  const s = secs % 60;

}

async function startLockIn() {
  if (!selectedMode) {
    alert('Select a lock mode first.');
    return;
  }

  const [tab] = await chrome.tabs.query({ active: true, currentWindow: true });
  if (!tab) return;

  const whitelistRaw = document.getElementById('lock-whitelist').value;
  const whitelist = whitelistRaw.split('\n').map(s => s.trim()).filter(Boolean);

  lockState = {
    active: true,
    mode: selectedMode,
    lockedTabId: tab.id,
    lockedWindowId: tab.windowId,
    startTime: Date.now(),
    whitelist,
  };

  saveLockState();
  showLockActiveView();

  lockTimerTick = setInterval(updateLockTimerDisplay, 1000);
}

function endLockIn() {
  const elapsed = getElapsedSecs();
  if (elapsed > 10) {
    // Confirm end
    const mins = Math.floor(elapsed / 60);
    if (!confirm(`End session? You've been focused for ${mins} minute${mins !== 1 ? 's' : ''}.`)) return;
  }

  // Record history
  lockHistory.unshift({
    mode: lockState.mode,
    duration: elapsed,
    startTime: lockState.startTime,
    date: new Date().toLocaleDateString('en-US', { month: 'short', day: 'numeric' })
  });
  if (lockHistory.length > 30) lockHistory = lockHistory.slice(0, 30);
  saveLockHistory();

  lockState = { active: false, mode: null, lockedTabId: null, lockedWindowId: null, startTime: null, whitelist: [] };
  saveLockState();

  clearInterval(lockTimerTick);
  lockTimerTick = null;

  showLockSelectorView();
  renderLockHistory();
}

function navigateTo(toolId) {
  const prev = views[currentView];
  const next = views[toolId];
  if (!next || toolId === currentView) return;

  // Fade out previous
  if (prev) {
    prev.style.opacity = '0';
    prev.style.transform = toolId === 'home' ? 'translateX(-8px)' : 'translateX(8px)';
    setTimeout(() => {
      prev.classList.remove('active');
      prev.style.opacity = '';
      prev.style.transform = '';
    }, 120);
  }

  // Show and fade in next
  setTimeout(() => {
    next.classList.add('active');
    next.style.opacity = '0';
    next.style.transform = toolId === 'home' ? 'translateX(8px)' : 'translateX(-8px)';
    requestAnimationFrame(() => {
      next.style.transition = 'opacity 0.12s ease, transform 0.12s ease';
      next.style.opacity = '1';
      next.style.transform = 'translateX(0)';
      setTimeout(() => {
        next.style.transition = '';
        next.style.opacity = '';
        next.style.transform = '';
      }, 130);
    });

    currentView = toolId;
    if (!toolInited[toolId]) {
      initTool(toolId);
      toolInited[toolId] = true;
    }
  }, prev ? 100 : 0);
}
